##### 点亮8*8点阵图上的一个小灯

```
#include<reg52.h>

sbit DIO=P3^4;	 //串行数据口
sbit S_CLK=P3^5; //移位寄存器时钟
sbit R_CLK=P3^6; //输出锁存器时钟

void main()
{
   //row 0x80, col 0xfe
   unsigned char i,dat;
   dat=0xfe;
   for(i=0;i<8;i++)
   {
   		S_CLK=0;
		R_CLK=0;
		if(dat&0x01)
			DIO=1;
		else
			DIO=0;
		S_CLK=1;
		dat>>=1;//dat右移一位		
   }
   dat=0x80;
   for(i=0;i<8;i++)
   {
   		S_CLK=0;
		R_CLK=0;
		if(dat&0x01)
			DIO=1;
		else
			DIO=0;
		S_CLK=1;
		dat>>=1;		
   }
   R_CLK=1;
   while(1);
}
```

<!--读取方式：串行输入，并行输出-->

<!--S_CLK,R_CLK 上升沿(0->1)时进行数据移位和输出-->
