##### 定时器（数码管显示）

*<u>定时器定时，自动改变数码管显示的数字</u>*

```
#include<reg52.h>

#define uint unsigned int
#define uchar unsigned char

sbit DU=P2^6;//数码管段选
sbit WE=P2^7;//数码管位选

//共阴极数码管0~9的段选表 
//code 其放在rom,不被改变
uchar code tabel[]={0x3f,0x06,0x5b,0x4f,0x66,
      0x6d,0x7d,0x07,0x7f,0x6f};

void delay(uint z)
{
	uint x,y;
	for(x=z;x>0;x--)
		for(y=114;y>0;y--);
}

void display(uchar i) //0~255
{
	uchar bai,shi,ge;
	bai=i/100;
	shi=i%100/10;
	ge=i%10;
    
    //第一位数码管显示
	P0=0xff;//清除断码
	WE=1;//打开位选锁存器
	P0=0xfe;//1111 1110 D7~D0 （亮的位送0）
	WE=0;//锁存位选数据

	DU=1; //打开段选锁存器
	P0=tabel[bai]; 
	DU=0;//锁存段选数据
	delay(5); //停留5毫秒
	
    //第二位数码管显示
	P0=0xff;//清除断码
	WE=1;//打开位选锁存器
	P0=0xfd;//1111 1101 D7~D0
	WE=0;//锁存位选数据

	DU=1; //打开段选锁存器
	P0=tabel[shi];
	DU=0;//锁存段选数据
	delay(5);

    //第三位数码管显示
    P0=0xff;//清除断码
	WE=1;//打开位选锁存器
	P0=0xfb;//1111 1011 D7~D0
	WE=0;//锁存位选数据

	DU=1; //打开段选锁存器
	P0=tabel[ge]; 
	DU=0;//锁存段选数据
	delay(5);

}

void timer0Init()
{
	TR0=1;//启动定时器0
	TMOD=0x01;//工作模式1，16位定时器
	//定时50ms，设初值
	TH0=0x4b;//(65535-46082)/256
	TL0=0xfd; //(65535-46082)%256
}

void main()
{
	uchar msec,sec;//定义毫秒和秒变量
	timer0Init();
	while(1)
	{
		 if(TF0==1)
		 {
		 	TF0=0;//由软件清零溢出标志位
			TH0=0x4b;
			TL0=0xfd; //重新设置初值
		 	msec++;//50ms到
			if(msec==20)
			{
				msec=0;
				sec++;//1s到
				if(sec==10)
					sec=0;//秒清零
			}		
		 }
		 display(sec);
	}
}
```
